[docs]	def formaturl(self, endpoint, subs):
		url, delta = self.url, 0
		subs = list(reversed(subs))
		for s in self.url_sub:
			val = subs.pop()
			url = url[:s[0]+delta] + val + url[s[1]+delta:]
			delta = len(val) - (s[1]-s[0])
		return endpoint + url
